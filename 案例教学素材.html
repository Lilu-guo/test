# 线性回归 version:2.1
import numpy as np
import matplotlib.pyplot as plt
m=20
data=[ 3,  4,  5,  4,  9,  7,  9, 11, 12, 15, 18, 18, 21, 19, 24, 22, 26, 25, 27, 30]
X0=np.ones((m,1))
X1=np.arange(1,m+1).reshape(m,1)
X=np.hstack((X0,X1))
y=np.array(data).reshape(m,1)
alpha=0.01
theta=[1,1]

# 计算损失
def error_function(theta, X, y):
    diff=np.dot(X,theta)-y
    return (1./(2*m)) * np.dot(np.transpose(diff), diff)

# 梯度公式
def gradient_function(theta, X, y):
    diff=np.dot(X,theta)-y
    return (1./m) * np.dot(np.transpose(X), diff)

# 优化迭代
def gradient_descent(X,y,alpha,theta):
    #theta初值
    theta=np.array(theta).reshape(2,1)
    gradient=gradient_function(theta,X,y)
    cout=0
    #while not np.all(np.absolute(gradient)<=1e-5):
    for i in np.arange(1,1e4+1):
        cout=cout+1
        theta=theta-alpha*gradient
        #更新梯度
        gradient=gradient_function(theta,X,y)
    return theta,gradient,cout

# 调用
optimal,gradient,cout=gradient_descent(X,y,alpha,theta)

print('theta:',optimal)
print('gradient:',gradient)
# 使用最优theta代入，求损失函数值
print('error function:',error_function(optimal,X,y))
print('cout:',cout)
